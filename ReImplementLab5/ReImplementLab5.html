                <script>
                function sumUsingReduce(arr) {
                    let r = arr.reduce(function (a, b) {
                        return a+b;
                      })
                    return r
                }
                
                function multiplyUsingReduce(arr) {
                    let r = arr.reduce(function (a, b) {
                        return a*b;
                      })
                    return r
                }
